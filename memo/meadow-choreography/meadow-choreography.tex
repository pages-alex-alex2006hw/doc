\paragraph{Example 1: Sum of squares}
\begin{verbatim}
set agents = {m1, m2, ..., m10};
set numbers = {1, 2, 3, ..., 10};
foreach u in agents fork {
  send_request(u, SQUARE_REQ, numbers[i]);
  get_result(u, SQUARE_REQ, results[i]);
} join_all;
for (int i = 0; i < 10; i++)
  sum += results[i];
\end{verbatim}
