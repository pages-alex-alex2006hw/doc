\paragraph{Linux Kernel 2.4.2 Compilation Procedure}
\ben
\w prepare \verb+/usr/src/linux+ directory
\w generate \verb+.config+ in {\tt /usr/src/linux}
\w {\tt make oldconfig}
\w {\tt make dep}
\w {\tt make clean}
\w {\tt make bzImage}
\w {\tt make modules}
\w {\tt cp /usr/src/linux/arch/i386/boot/bzImage /boot/vmlinuz.new}
\w {\tt cp /usr/src/linux/System.map /boot/System.map.new}
\w {\tt cd boot}
	\bit
	\w {\tt rm /boot/vmlinuz}
	\w {\tt rm /boot/System.map}
	\w {\tt ln -s /boot/System.map.new /boot/System.map}
	\w {\tt ln -s /boot/vmlinuz.new /boot/vmlinuz}
	\eit
\w add entry to the {\tt etc/lilo.conf}
\w {\tt /sbin/lilo}
\w REBOOT
\een
